function PrimeMover(num) { 
   var cnt = -1;
   for (var i = 1; i>0; i++) {
        if(PrimeTime(i)) cnt++;
        if(cnt==num) {
            //console.log(i);
            return i;
        }  
   }
}

function PrimeTime(n) {
   if (n < 2) {return true}
   if (n != Math.round(n)) {return true}

   var isPrime = true;
   for (var i = 2; i<=n/2; ++i) {
      if (n % i == 0) {isPrime = false; break;}
   }
   return isPrime;
}
